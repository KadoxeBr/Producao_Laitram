const URL_API = "https://script.google.com/macros/s/AKfycbx3B4neQ7N6Nb8e6-DFOodN0nDfCXWoJ5bpWdX-vChSYarn6U5Auvh2trQS-wFODSEw/exec"; // Configuração de datas ao carregar const hoje = new Date().toISOString().split('T')[0]; document.getElementById('dataProd').value = hoje; document.getElementById('filtroInicio').value = hoje; document.getElementById('filtroFim').value = hoje; // SALVAR DADOS document.getElementById('prodForm').addEventListener('submit', async function(e) { e.preventDefault(); const btn = document.getElementById('btnSalvar'); btn.innerText = "⏳ ENVIANDO..."; btn.disabled = true; const pIn = parseFloat(document.getElementById('pesoIn').value) || 0; const pFile = parseFloat(document.getElementById('pesoOutFile').value) || 0; const pBroken = parseFloat(document.getElementById('pesoOutQuebrado').value) || 0; // Cálculo local antes de enviar const totalSaida = pFile + pBroken; const percPerda = pIn > 0 ? (((pIn - totalSaida) / pIn) * 100).toFixed(2) : "0.00"; const dados = { data: document.getElementById('dataProd').value, lote: document.getElementById('lote').value, turno: document.getElementById('turno').value, calibre: document.getElementById('calibre').value, entrada: pIn, file: pFile, quebrado: pBroken, perda: percPerda // Enviamos o cálculo pronto }; try { await fetch(URL_API, { method: 'POST', mode: 'no-cors', body: JSON.stringify(dados) }); alert("✅ Registro salvo com sucesso!"); this.reset(); document.getElementById('dataProd').value = hoje; atualizarInterface(); } catch (err) { alert("❌ Erro de conexão."); } finally { btn.innerText = "Confirmar e Enviar"; btn.disabled = false; } }); // BUSCAR E CALCULAR (O segredo da correção está aqui) async function atualizarInterface() { const loading = document.getElementById('loadingStatus'); if(loading) loading.classList.remove('hidden'); const corpo = document.getElementById('tabelaCorpo'); const inicio = document.getElementById('filtroInicio').value; const fim = document.getElementById('filtroFim').value; try { const response = await fetch(URL_API + "?t=" + new Date().getTime()); const rawData = await response.json(); // MAPEAMENTO ROBUSTO: Forçamos a conversão para número e recalculamos a perda no ato da leitura const dadosProcessados = rawData.map(r => { const ent = parseFloat(r[4]) || 0; const fil = parseFloat(r[5]) || 0; const queb = parseFloat(r[6]) || 0; const calcPerda = ent > 0 ? (((ent - (fil + queb)) / ent) * 100).toFixed(2) : "0.00"; return { data: r[0], lote: r[1], turno: r[2], calibre: r[3], entrada: ent, file: fil, quebrado: queb, perda: calcPerda }; }); corpo.innerHTML = ""; const filtrados = dadosProcessados.filter(d => { const dData = String(d.data).split('T')[0]; return dData >= inicio && dData <= fim; }); filtrados.reverse().forEach(reg => { const dataF = String(reg.data).split('T')[0].split('-').reverse().join('/'); corpo.innerHTML += ` ${dataF} ${reg.lote} ${reg.turno} ${reg.calibre} ${reg.entrada.toFixed(2)} ${reg.file.toFixed(2)} ${reg.quebrado.toFixed(2)} ${reg.perda}% `; }); renderizarDashboard(filtrados); } catch (error) { console.error(error); corpo.innerHTML = "ERRO DE SINCRONIZAÇÃO"; } finally { if(loading) loading.classList.add('hidden'); } } function renderizarDashboard(lista) { const calc = (arr) => { const sum = arr.reduce((acc, c) => { acc.in += c.entrada; acc.file += c.file; acc.broken += c.quebrado; return acc; }, {in:0, file:0, broken:0}); const lossKg = sum.in - (sum.file + sum.broken); return { in: sum.in.toFixed(2), file: sum.file.toFixed(2), brokenKg: sum.broken.toFixed(2), pLoss: sum.in > 0 ? ((lossKg/sum.in)*100).toFixed(1) : "0.0", pBroken: sum.in > 0 ? ((sum.broken/sum.in)*100).toFixed(1) : "0.0" }; }; const g = calc(lista); const d = calc(lista.filter(x => x.turno === 'Dia')); const n = calc(lista.filter(x => x.turno === 'Noite')); const atualizarCard = (pref, obj) => { document.getElementById(pref + 'In').innerText = obj.in; document.getElementById(pref + 'File').innerText = obj.file; document.getElementById(pref + 'BrokenKg').innerText = obj.brokenKg; document.getElementById(pref + 'Perda').innerText = obj.pLoss + "%"; document.getElementById(pref + 'QuebradoPerc').innerText = obj.pBroken + "%"; }; atualizarCard('g', g); atualizarCard('d', d); atualizarCard('n', n); } atualizarInterface();
